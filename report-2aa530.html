152/* Convert a string into a signed 64 bit integer.
153 * The function returns 1 if the string could be parsed into a (non-overflowing)

156 *
157 * Note that this function demands that the string strictly represents
158 * a int64 value: no spaces or other characters before or after the string
159 * representing the number are accepted, nor zeroes at the start if not
160 * for the string "0" representing the zero number.
161 *
162 * Because of its strictness, it is safe to use this function to check if

177 const char *p = s;
178 unsigned long plen = 0;
179 int negative = 0;
180 uint64_t v;
181
182 /* Abort if length indicates this cannot possibly be an int */
183 if (slen == 0 || slen >= LONG_STR_SIZE21)
184 return 0;
185
186 /* Special case: first and only digit is 0. */
187 if (slen == 1 && p[0] == '0') {
188 if (value != NULL((void*)0)) *value = 0;
189 return 1;
190 }
191
192 if (p[0] == '-') {
193 negative = 1;
194 p++; plen++;
195
196 /* Abort on only a negative sign. */
197 if (plen == slen)
198 return 0;
199 }
200
201 /* First digit should be 1-9, otherwise the string should just be 0. */
202 if (p[0] >= '1' && p[0] <= '9') {
203 v = p[0]-'0';
204 p++; plen++;
205 } else {
206 return 0;
207 }
208
209 while (plen < slen && p[0] >= '0' && p[0] <= '9') {
210 if (v > (UINT64_MAX(18446744073709551615UL) / 10)) /* Overflow. */
211 return 0;
212 v *= 10;

214 if (v > (UINT64_MAX(18446744073709551615UL) - (p[0]-'0'))) /* Overflow. */
215 return 0;
216 v += p[0]-'0';
217
218 p++; plen++;
219 }

223 return 0;

231 return 0;

482/* If 'p' points to an element of the listpack, calling lpPrev() will return
483 * the pointer to the previous element (the one on the left), or NULL if 'p'
484 * already pointed to the first element of the listpack. */
485unsigned char *lpPrev(unsigned char *lp, unsigned char *p) {
486 assert(p)(__builtin_expect(!!((p)), 1)?(void)0 : (_serverAssert("p","listpack.c"
,486),__builtin_unreachable()));
487 if (p-lp == LP_HDR_SIZE6) return NULL((void*)0);
488 p--; /* Seek the first backlen byte of the last element. */
489 uint64_t prevlen = lpDecodeBacklen(p);
490 prevlen += lpEncodeBacklen(NULL((void*)0),prevlen);
491 p -= prevlen-1; /* Seek the first byte of the previous entry. */
492 lpAssertValidEntry(lp, lpBytes(lp), p);
493 return p;
494}

513 * to use the cached value when within range, otherwise a full scan is
514 * needed. As a side effect of calling this function, the listpack header
515 * could be modified, because if the count is found to be already within
516 * the 'numele' header field range, the new value is set. */

536/* Return the listpack element pointed by 'p'.
537 *
538 * The function changes behavior depending on the passed 'intbuf' value.
539 * Specifically, if 'intbuf' is NULL:
540 *
541 * If the element is internally encoded as an integer, the function returns
542 * NULL and populates the integer value by reference in 'count'. Otherwise if
543 * the element is encoded as a string a pointer to the string (pointing inside
544 * the listpack itself) is returned, and 'count' is set to the length of the
545 * string.
546 *
547 * If instead 'intbuf' points to a buffer passed by the caller, that must be
548 * at least LP_INTBUF_SIZE bytes, the function always returns the element as
549 * it was a string (returning the pointer to the string and setting the
550 * 'count' argument to the string length by reference). However if the element
551 * is encoded as an integer, the 'intbuf' buffer is used in order to store
552 * the string representation.
553 *
554 * The user should use one or the other form depending on what the value will
555 * be used for. If there is immediate usage for an integer value returned
556 * by the function, than to pass a buffer (and convert it back to a number)
557 * is of course useless.
558 *
559 * If 'entry_size' is not NULL, *entry_size is set to the entry length of the
560 * listpack element pointed by 'p'. This includes the encoding bytes, length
561 * bytes, the element data itself, and the backlen bytes.
562 *
563 * If the function is called against a badly encoded ziplist, so that there
564 * is no valid way to parse it, the function returns like if there was an
565 * integer encoded with value 12345678900000000 + <unrecognized byte>, this may
566 * be an hint to understand that something is wrong. To crash in this case is
567 * not sensible because of the different requirements of the application using
568 * this lib.
569 *
570 * Similarly, there is no error returned since the listpack normally can be
571 * assumed to be valid, so that would be a very high API cost. */
572static inline unsigned char *lpGetWithSize(unsigned char *p, int64_t *count, unsigned char *intbuf, uint64_t *entry_size) {
573 int64_t val;
574 uint64_t uval, negstart, negmax;
575
576 assert(p)(__builtin_expect(!!((p)), 1)?(void)0 : (_serverAssert("p","listpack.c"
,576),__builtin_unreachable())); /* assertion for valgrind (avoid NPD) */
577 if (LP_ENCODING_IS_7BIT_UINT(p[0])(((p[0])&0x80)==0)) {
578 negstart = UINT64_MAX(18446744073709551615UL); /* 7 bit ints are always positive. */
579 negmax = 0;
580 uval = p[0] & 0x7f;
581 if (entry_size) *entry_size = LP_ENCODING_7BIT_UINT_ENTRY_SIZE2;
582 } else if (LP_ENCODING_IS_6BIT_STR(p[0])(((p[0])&0xC0)==0x80)) {

591 } else if (LP_ENCODING_IS_16BIT_INT(p[0])(((p[0])&0xFF)==0xF1)) {

595 negmax = UINT16_MAX(65535);

597 } else if (LP_ENCODING_IS_24BIT_INT(p[0])(((p[0])&0xFF)==0xF2)) {
598 uval = (uint64_t)p[1] |
599 (uint64_t)p[2]<<8 |
600 (uint64_t)p[3]<<16;
601 negstart = (uint64_t)1<<23;
602 negmax = UINT32_MAX(4294967295U)>>8;
603 if (entry_size) *entry_size = LP_ENCODING_24BIT_INT_ENTRY_SIZE5;
604 } else if (LP_ENCODING_IS_32BIT_INT(p[0])(((p[0])&0xFF)==0xF3)) {
605 uval = (uint64_t)p[1] |
606 (uint64_t)p[2]<<8 |

608 (uint64_t)p[4]<<24;
609 negstart = (uint64_t)1<<31;
610 negmax = UINT32_MAX(4294967295U);
611 if (entry_size) *entry_size = LP_ENCODING_32BIT_INT_ENTRY_SIZE6;
612 } else if (LP_ENCODING_IS_64BIT_INT(p[0])(((p[0])&0xFF)==0xF4)) {
613 uval = (uint64_t)p[1] |
614 (uint64_t)p[2]<<8 |
615 (uint64_t)p[3]<<16 |
616 (uint64_t)p[4]<<24 |
617 (uint64_t)p[5]<<32 |
618 (uint64_t)p[6]<<40 |
619 (uint64_t)p[7]<<48 |
620 (uint64_t)p[8]<<56;
621 negstart = (uint64_t)1<<63;
622 negmax = UINT64_MAX(18446744073709551615UL);
623 if (entry_size) *entry_size = LP_ENCODING_64BIT_INT_ENTRY_SIZE10;
624 } else if (LP_ENCODING_IS_12BIT_STR(p[0])(((p[0])&0xF0)==0xE0)) {
625 *count = LP_ENCODING_12BIT_STR_LEN(p)((((p)[0] & 0xF) << 8) | (p)[1]);
626 if (entry_size) *entry_size = 2 + *count + lpEncodeBacklen(NULL((void*)0), *count + 2);
627 return p+2;
628 } else if (LP_ENCODING_IS_32BIT_STR(p[0])(((p[0])&0xFF)==0xF0)) {
629 *count = LP_ENCODING_32BIT_STR_LEN(p)(((uint32_t)(p)[1]<<0) | ((uint32_t)(p)[2]<<8) | (
(uint32_t)(p)[3]<<16) | ((uint32_t)(p)[4]<<24));
630 if (entry_size) *entry_size = 5 + *count + lpEncodeBacklen(NULL((void*)0), *count + 5);
631 return p+5;
632 } else {
633 uval = 12345678900000000ULL + p[0];
634 negstart = UINT64_MAX(18446744073709551615UL);
635 negmax = 0;
636 }
637
638 /* We reach this code path only for integer encodings.
639 * Convert the unsigned value to the signed one using two's complement
640 * rule. */
641 if (uval >= negstart) {
642 /* This three steps conversion should avoid undefined behaviors
643 * in the unsigned -> signed conversion. */
644 uval = negmax-uval;
645 val = uval;
646 val = -val-1;
647 } else {
648 val = uval;
649 }
650
651 /* Return the string representation of the integer or the value itself
652 * depending on intbuf being NULL or not. */
653 if (intbuf) {
654 *count = ll2string((char*)intbuf,LP_INTBUF_SIZE21,(long long)val);
655 return intbuf;
656 } else {
657 *count = val;
658 return NULL((void*)0);
659 }
660}

684/* Find pointer to the entry equal to the specified entry. Skip 'skip' entries
685 * between every comparison. Returns NULL when the field could not be found. */
686unsigned char *lpFind(unsigned char *lp, unsigned char *p, unsigned char *s,
687 uint32_t slen, unsigned int skip) {
688 int skipcnt = 0;
689 unsigned char vencoding = 0;
690 unsigned char *value;
691 int64_t ll, vll;
692 uint64_t entry_size = 123456789; /* initialized to avoid warning. */
693 uint32_t lp_bytes = lpBytes(lp);
694
695 assert(p)(__builtin_expect(!!((p)), 1)?(void)0 : (_serverAssert("p","listpack.c"
,695),__builtin_unreachable()));
696 while (p) {
697 if (skipcnt == 0) {
698 value = lpGetWithSize(p, &ll, NULL((void*)0), &entry_size);
699 if (value) {
700 /* check the value doesn't reach outside the listpack before accessing it */
701 assert(p >= lp + LP_HDR_SIZE && p + entry_size < lp + lp_bytes)(__builtin_expect(!!((p >= lp + 6 && p + entry_size
< lp + lp_bytes)), 1)?(void)0 : (_serverAssert("p >= lp + LP_HDR_SIZE && p + entry_size < lp + lp_bytes"
,"listpack.c",701),__builtin_unreachable()));
702 if (slen == ll && memcmp(value, s, slen) == 0) {
703 return p;
704 }
705 } else {
706 /* Find out if the searched field can be encoded. Note that
707 * we do it only the first time, once done vencoding is set
708 * to non-zero and vll is set to the integer value. */
709 if (vencoding == 0) {
710 /* If the entry can be encoded as integer we set it to
711 * 1, else set it to UCHAR_MAX, so that we don't retry
712 * again the next time. */
713 if (slen >= 32 || slen == 0 || !lpStringToInt64((const char*)s, slen, &vll)) {
714 vencoding = UCHAR_MAX(127*2 +1);
715 } else {
716 vencoding = 1;
717 }
718 }
719
720 /* Compare current entry with specified entry, do it only
721 * if vencoding != UCHAR_MAX because if there is no encoding
722 * possible for the field it can't be a valid integer. */
723 if (vencoding != UCHAR_MAX(127*2 +1) && ll == vll) {
724 return p;
725 }
726 }
727
728 /* Reset skip count */
729 skipcnt = skip;
730 p += entry_size;
731 } else {
732 /* Skip entry */
733 skipcnt--;
734
735 /* Move to next entry, avoid use `lpNext` due to `ASSERT_INTEGRITY` in
736 * `lpNext` will call `lpBytes`, will cause performance degradation */
737 p = lpSkip(p);
738 }
739
740 /* The next call to lpGetWithSize could read at most 8 bytes past `p`
741 * We use the slower validation call only when necessary. */
742 if (p + 8 >= lp + lp_bytes)
743 lpAssertValidEntry(lp, lp_bytes, p);
744 else
745 assert(p >= lp + LP_HDR_SIZE && p < lp + lp_bytes)(__builtin_expect(!!((p >= lp + 6 && p < lp + lp_bytes
)), 1)?(void)0 : (_serverAssert("p >= lp + LP_HDR_SIZE && p < lp + lp_bytes"
,"listpack.c",745),__builtin_unreachable()));
746 if (p[0] == LP_EOF0xFF) break;
747 }
748
749 return NULL((void*)0);
750}

752/* Insert, delete or replace the specified string element 'elestr' of length
753 * 'size' or integer element 'eleint' at the specified position 'p', with 'p'
754 * being a listpack element pointer obtained with lpFirst(), lpLast(), lpNext(),
755 * lpPrev() or lpSeek().
756 *
757 * The element is inserted before, after, or replaces the element pointed
758 * by 'p' depending on the 'where' argument, that can be LP_BEFORE, LP_AFTER
759 * or LP_REPLACE.
760 *
761 * If both 'elestr' and `eleint` are NULL, the function removes the element
762 * pointed by 'p' instead of inserting one.
763 * If `eleint` is non-NULL, 'size' is the length of 'eleint', the function insert
764 * or replace with a 64 bit integer, which is stored in the 'eleint' buffer.
765 * If 'elestr` is non-NULL, 'size' is the length of 'elestr', the function insert
766 * or replace with a string, which is stored in the 'elestr' buffer.
767 *
768 * Returns NULL on out of memory or when the listpack total length would exceed
769 * the max allowed size of 2^32-1, otherwise the new pointer to the listpack
770 * holding the new element is returned (and the old pointer passed is no longer
771 * considered valid)
772 *
773 * If 'newp' is not NULL, at the end of a successful call '*newp' will be set
774 * to the address of the element just added, so that it will be possible to
775 * continue an interaction with lpNext() and lpPrev().
776 *
777 * For deletion operations (both 'elestr' and 'eleint' set to NULL) 'newp' is
778 * set to the next element, on the right of the deleted one, or to NULL if the
779 * deleted element was the last one. */

790 * zero-length element. So whatever we get passed as 'where', set
791 * it to LP_REPLACE. */
792 if (delete) where = LP_REPLACE2;
793
794 /* If we need to insert after the current element, we just jump to the
795 * next element (that could be the EOF one) and handle the case of
796 * inserting before. So the function will actually deal with just two
797 * cases: LP_BEFORE and LP_REPLACE. */
798 if (where == LP_AFTER1) {

800 where = LP_BEFORE0;
801 ASSERT_INTEGRITY(lp, p)do { (__builtin_expect(!!(((p) >= (lp)+6 && (p) <

24)))), 1)?(void)0 : (_serverAssert("(p) >= (lp)+LP_HDR_SIZE && (p) < (lp)+lpGetTotalBytes((lp))"
,"listpack.c",801),__builtin_unreachable())); } while (0);
802 }
803
804 /* Store the offset of the element 'p', so that we can obtain its
805 * address again after a reallocation. */
806 unsigned long poff = p-lp;
807
808 int enctype;

811 * element to be stored into 'intenc' in case it is representable as
812 * an integer: in that case, the function returns LP_ENCODING_INT.
813 * Otherwise if LP_ENCODING_STR is returned, we'll have to call
814 * lpEncodeString() to actually write the encoded string on place later.
815 *
816 * Whatever the returned encoding is, 'enclen' is populated with the
817 * length of the encoded element. */
818 enctype = lpEncodeGetType(elestr,size,intenc,&enclen);
819 if (enctype == LP_ENCODING_INT0) eleint = intenc;
820 } else if (eleint) {

822 enclen = size; /* 'size' is the length of the encoded integer element. */
823 } else {
824 enctype = -1;
825 enclen = 0;
826 }

829 * and append it to the end: this allows to traverse the listpack from
830 * the end to the start. */

842 if (new_listpack_bytes > UINT32_MAX(4294967295U)) return NULL((void*)0);

845 * allocation (in case 'when' value is LP_REPLACE and the new element is
846 * smaller). However we do that before memmoving the memory to
847 * make room for the new element if the final allocation will get
848 * larger, or we do it after if the final allocation will get smaller. */

1166/* Seek the specified element and returns the pointer to the seeked element.
1167 * Positive indexes specify the zero-based element to seek from the head to
1168 * the tail, negative indexes specify elements starting from the tail, where
1169 * -1 means the last element, -2 the penultimate and so forth. If the index
1170 * is out of range, NULL is returned. */
1171unsigned char *lpSeek(unsigned char *lp, long index) {
1172 int forward = 1; /* Seek forward by default. */
1173
1174 /* We want to seek from left to right or the other way around
1175 * depending on the listpack length and the element position.
1176 * However if the listpack length cannot be obtained in constant time,
1177 * we always seek from left to right. */
1178 uint32_t numele = lpGetNumElements(lp)(((uint32_t)(lp)[4]<<0) | ((uint32_t)(lp)[5]<<8));
1179 if (numele != LP_HDR_NUMELE_UNKNOWN(65535)) {
1180 if (index < 0) index = (long)numele+index;
1181 if (index < 0) return NULL((void*)0); /* Index still < 0 means out of range. */
1182 if (index >= (long)numele) return NULL((void*)0); /* Out of range the other side. */
1183 /* We want to scan right-to-left if the element we are looking for
1184 * is past the half of the listpack. */
1185 if (index > (long)numele/2) {
1186 forward = 0;
1187 /* Right to left scanning always expects a negative index. Convert
1188 * our index to negative form. */
1189 index -= numele;
1190 }
1191 } else {
1192 /* If the listpack length is unspecified, for negative indexes we
1193 * want to always scan right-to-left. */
1194 if (index < 0) forward = 0;
1195 }
1196
1197 /* Forward and backward scanning is trivially based on lpNext()/lpPrev(). */
1198 if (forward) {
1199 unsigned char *ele = lpFirst(lp);
1200 while (index > 0 && ele) {
1201 ele = lpNext(lp,ele);
1202 index--;
1203 }
1204 return ele;
1205 } else {
1206 unsigned char *ele = lpLast(lp);
1207 while (index < -1 && ele) {
1208 ele = lpPrev(lp,ele);
1209 index++;
1210 }
1211 return ele;
1212 }
1213}

1222/* Validate the integrity of a single listpack entry and move to the next one.
1223 * The input argument 'pp' is a reference to the current record and is advanced on exit.
1224 * Returns 1 if valid, 0 if invalid. */

1227 (p) < lp + LP_HDR_SIZE6 || \
1228 (p) > lp + lpbytes - 1)

1230 if (!p)
1231 return 0;

1235 return 0;

1238 *pp = NULL((void*)0);
1239 return 1;

1243 uint32_t lenbytes = lpCurrentEncodedSizeBytes(p);

1248 if (OUT_OF_RANGE(p + lenbytes))

1278/* Validate the integrity of the data structure.
1279 * when `deep` is 0, only the integrity of the header is validated.
1280 * when `deep` is 1, we scan all the entries one by one. */
1281int lpValidateIntegrity(unsigned char *lp, size_t size, int deep,
1282 listpackValidateEntryCB entry_cb, void *cb_userdata) {
1283 /* Check that we can actually read the header. (and EOF) */
1284 if (size < LP_HDR_SIZE6 + 1)
1285 return 0;
1286
1287 /* Check that the encoded size in the header must match the allocated size. */
1288 size_t bytes = lpGetTotalBytes(lp)(((uint32_t)(lp)[0]<<0) | ((uint32_t)(lp)[1]<<8) |
((uint32_t)(lp)[2]<<16) | ((uint32_t)(lp)[3]<<24
));
1289 if (bytes != size)
1290 return 0;
1291
1292 /* The last byte must be the terminator. */
1293 if (lp[size-1] != LP_EOF0xFF)
1294 return 0;
1295
1296 if (!deep)
1297 return 1;
1298
1299 /* Validate the individual entries. */
1300 uint32_t count = 0;
1301 uint32_t numele = lpGetNumElements(lp)(((uint32_t)(lp)[4]<<0) | ((uint32_t)(lp)[5]<<8));
1302 unsigned char *p = lp + LP_HDR_SIZE6;
1303 while(p && p[0] != LP_EOF0xFF) {
1304 unsigned char *prev = p;
1305
1306 /* Validate this entry and move to the next entry in advance
1307 * to avoid callback crash due to corrupt listpack. */
1308 if (!lpValidateNext(lp, &p, bytes))
1309 return 0;
1310
1311 /* Optionally let the caller validate the entry too. */
1312 if (entry_cb && !entry_cb(prev, numele, cb_userdata))
1313 return 0;
1314
1315 count++;
1316 }
1317
1318 /* Make sure 'p' really does point to the end of the listpack. */
1319 if (p != lp + size - 1)
1320 return 0;
1321
1322 /* Check that the count in the header is correct */
1323 if (numele != LP_HDR_NUMELE_UNKNOWN(65535) && numele != count)
1324 return 0;
1325
1326 return 1;
1327}

1384/* Randomly select count of key value pairs and store into 'keys' and
1385 * 'vals' args. The order of the picked entries is random, and the selections
1386 * are non-unique (repetitions are possible).
1387 * The 'vals' arg can be NULL in which case we skip these. */
1388void lpRandomPairs(unsigned char *lp, unsigned int count, listpackEntry *keys, listpackEntry *vals) {
1389 unsigned char *p, *key, *value;
1390 unsigned int klen = 0, vlen = 0;
1391 long long klval = 0, vlval = 0;
1392
1393 /* Notice: the index member must be first due to the use in uintCompare */
1394 typedef struct {
1395 unsigned int index;
1396 unsigned int order;
1397 } rand_pick;
1398 rand_pick *picks = zmalloc(sizeof(rand_pick)*count);
1399 unsigned int total_size = lpLength(lp)/2;
1400
1401 /* Avoid div by zero on corrupt listpack */
1402 assert(total_size)(__builtin_expect(!!((total_size)), 1)?(void)0 : (_serverAssert
("total_size","listpack.c",1402),__builtin_unreachable()));
1403
1404 /* create a pool of random indexes (some may be duplicate). */
1405 for (unsigned int i = 0; i < count; i++) {
1406 picks[i].index = (rand() % total_size) * 2; /* Generate even indexes */
1407 /* keep track of the order we picked them */
1408 picks[i].order = i;
1409 }
1410
1411 /* sort by indexes. */
1412 qsort(picks, count, sizeof(rand_pick), uintCompare);
1413
1414 /* fetch the elements form the listpack into a output array respecting the original order. */
1415 unsigned int lpindex = picks[0].index, pickindex = 0;
1416 p = lpSeek(lp, lpindex);
1417 while (p && pickindex < count) {
1418 key = lpGetValue(p, &klen, &klval);
1419 assert((p = lpNext(lp, p)))(__builtin_expect(!!(((p = lpNext(lp, p)))), 1)?(void)0 : (_serverAssert
("(p = lpNext(lp, p))","listpack.c",1419),__builtin_unreachable
()));
1420 value = lpGetValue(p, &vlen, &vlval);
1421 while (pickindex < count && lpindex == picks[pickindex].index) {
1422 int storeorder = picks[pickindex].order;
1423 lpSaveValue(key, klen, klval, &keys[storeorder]);
1424 if (vals)
1425 lpSaveValue(value, vlen, vlval, &vals[storeorder]);
1426 pickindex++;
1427 }
1428 lpindex += 2;
1429 p = lpNext(lp, p);
1430 }
1431
1432 zfree(picks);
1433}

1435/* Randomly select count of key value pairs and store into 'keys' and
1436 * 'vals' args. The selections are unique (no repetitions), and the order of
1437 * the picked entries is NOT-random.
1438 * The 'vals' arg can be NULL in which case we skip these.
1439 * The return value is the number of items picked which can be lower than the
1440 * requested count if the listpack doesn't hold enough pairs. */
1441unsigned int lpRandomPairsUnique(unsigned char *lp, unsigned int count, listpackEntry *keys, listpackEntry *vals) {
1442 unsigned char *p, *key;
1443 unsigned int klen = 0;
1444 long long klval = 0;
1445 unsigned int total_size = lpLength(lp)/2;
1446 unsigned int index = 0;
1447 if (count > total_size)
1448 count = total_size;
1449
1450 /* To only iterate once, every time we try to pick a member, the probability
1451 * we pick it is the quotient of the count left we want to pick and the
1452 * count still we haven't visited in the dict, this way, we could make every
1453 * member be equally picked.*/
1454 p = lpFirst(lp);
1455 unsigned int picked = 0, remaining = count;
1456 while (picked < count && p) {
1457 double randomDouble = ((double)rand()) / RAND_MAX2147483647;
1458 double threshold = ((double)remaining) / (total_size - index);
1459 if (randomDouble <= threshold) {
1460 key = lpGetValue(p, &klen, &klval);
1461 lpSaveValue(key, klen, klval, &keys[picked]);
1462 assert((p = lpNext(lp, p)))(__builtin_expect(!!(((p = lpNext(lp, p)))), 1)?(void)0 : (_serverAssert
("(p = lpNext(lp, p))","listpack.c",1462),__builtin_unreachable
()));
1463 if (vals) {
1464 key = lpGetValue(p, &klen, &klval);
1465 lpSaveValue(key, klen, klval, &vals[picked]);
1466 }
1467 remaining--;
1468 picked++;
1469 } else {
1470 assert((p = lpNext(lp, p)))(__builtin_expect(!!(((p = lpNext(lp, p)))), 1)?(void)0 : (_serverAssert
("(p = lpNext(lp, p))","listpack.c",1470),__builtin_unreachable
()));
1471 }
1472 p = lpNext(lp, p);
1473 index++;
1474 }
1475 return picked;
1476}
