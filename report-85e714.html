140/* The link to a sentinelRedisInstance. When we have the same set of Sentinels
141 * monitoring many masters, we have different instances representing the
142 * same Sentinels, one per master, and we need to share the hiredis connections
143 * among them. Otherwise if 5 Sentinels are monitoring 100 masters we create
144 * 500 outgoing connections instead of 5.
145 *
146 * So this structure represents a reference counted link in terms of the two
147 * hiredis connections for commands and Pub/Sub, and the fields needed for
148 * failure detection, since the ping/pong time are now local to the link: if
149 * the link is available, the instance is available. This way we don't just
150 * have 5 connections instead of 500, we also send 5 pings instead of 500.
151 *
152 * Links are shared only for Sentinels: master and slave instances have
153 * a link with refcount = 1, always. */
154typedef struct instanceLink {
155 int refcount; /* Number of sentinelRedisInstance owners. */
156 int disconnected; /* Non-zero if we need to reconnect cc or pc. */
157 int pending_commands; /* Number of commands sent waiting for a reply. */
158 redisAsyncContext *cc; /* Hiredis context for commands. */
159 redisAsyncContext *pc; /* Hiredis context for Pub / Sub. */
160 mstime_t cc_conn_time; /* cc connection time. */
161 mstime_t pc_conn_time; /* pc connection time. */
162 mstime_t pc_last_activity; /* Last time we received any message. */
163 mstime_t last_avail_time; /* Last time the instance replied to ping with
164 a reply we consider valid. */
165 mstime_t act_ping_time; /* Time at which the last pending ping (no pong
166 received after it) was sent. This field is
167 set to 0 when a pong is received, and set again
168 to the current time if the value is 0 and a new
169 ping is sent. */
170 mstime_t last_ping_time; /* Time at which we sent the last ping. This is
171 only used to avoid sending too many pings
172 during failure. Idle time is computed using
173 the act_ping_time field. */
174 mstime_t last_pong_time; /* Last time the instance replied to ping,
175 whatever the reply was. That's used to check
176 if the link is idle and must be reconnected. */
177 mstime_t last_reconn_time; /* Last reconnection attempt performed when
178 the link was down. */
179} instanceLink;

360static int redisAeAttach(aeEventLoop *loop, redisAsyncContext *ac) {
361 redisContext *c = &(ac->c);
362 redisAeEvents *e;
363
364 /* Nothing should be attached when something is already attached */
365 if (ac->ev.data != NULL((void*)0))
366 return C_ERR-1;
367
368 /* Create container for context and r/w events */
369 e = (redisAeEvents*)zmalloc(sizeof(*e));
370 e->context = ac;
371 e->loop = loop;
372 e->fd = c->fd;
373 e->reading = e->writing = 0;
374
375 /* Register functions to start/stop listening for events */
376 ac->ev.addRead = redisAeAddRead;
377 ac->ev.delRead = redisAeDelRead;
378 ac->ev.addWrite = redisAeAddWrite;
379 ac->ev.delWrite = redisAeDelWrite;
380 ac->ev.cleanup = redisAeCleanup;
381 ac->ev.data = e;
382
383 return C_OK0;

814/* Run pending scripts if we are not already at max number of running
815 * scripts. */
816void sentinelRunPendingScripts(void) {
817 listNode *ln;
818 listIter li;
819 mstime_t now = mstime();
820
821 /* Find jobs that are not running and run them, from the top to the
822 * tail of the queue, so we run older jobs first. */
823 listRewind(sentinel.scripts_queue,&li);
824 while (sentinel.running_scripts < SENTINEL_SCRIPT_MAX_RUNNING16 &&
825 (ln = listNext(&li)) != NULL((void*)0))
826 {
827 sentinelScriptJob *sj = ln->value;
828 pid_t pid;
829
830 /* Skip if already running. */
831 if (sj->flags & SENTINEL_SCRIPT_RUNNING1) continue;
832
833 /* Skip if it's a retry, but not enough time has elapsed. */
834 if (sj->start_time && sj->start_time > now) continue;
835
836 sj->flags |= SENTINEL_SCRIPT_RUNNING1;
837 sj->start_time = mstime();
838 sj->retry_num++;
839 pid = fork();
840
841 if (pid == -1) {
842 /* Parent (fork error).
843 * We report fork errors as signal 99, in order to unify the
844 * reporting with other kind of errors. */
845 sentinelEvent(LL_WARNING3,"-script-error",NULL((void*)0),
846 "%s %d %d", sj->argv[0], 99, 0);
847 sj->flags &= ~SENTINEL_SCRIPT_RUNNING1;
848 sj->pid = 0;
849 } else if (pid == 0) {
850 /* Child */
851 tlsCleanup();
852 execve(sj->argv[0],sj->argv,environ);
853 /* If we are here an error occurred. */
854 _exit(2); /* Don't retry execution. */
855 } else {
856 sentinel.running_scripts++;
857 sj->pid = pid;
858 sentinelEvent(LL_DEBUG0,"+script-child",NULL((void*)0),"%ld",(long)pid);
859 }
860 }
861}

863/* How much to delay the execution of a script that we need to retry after
864 * an error?
865 *
866 * We double the retry delay for every further retry we do. So for instance
867 * if RETRY_DELAY is set to 30 seconds and the max number of retries is 10
868 * starting from the second attempt to execute the script the delays are:
869 * 30 sec, 60 sec, 2 min, 4 min, 8 min, 16 min, 32 min, 64 min, 128 min. */
870mstime_t sentinelScriptRetryDelay(int retry_num) {
871 mstime_t delay = sentinel_script_retry_delay;
872
873 while (retry_num-- > 1) delay *= 2;
874 return delay;
875}

877/* Check for scripts that terminated, and remove them from the queue if the
878 * script terminated successfully. If instead the script was terminated by
879 * a signal, or returned exit code "1", it is scheduled to run again if
880 * the max number of retries did not already elapsed. */
881void sentinelCollectTerminatedScripts(void) {
882 int statloc;
883 pid_t pid;
884
885 while ((pid = waitpid(-1, &statloc, WNOHANG1)) > 0) {
886 int exitcode = WEXITSTATUS(statloc)(((statloc) & 0xff00) >> 8);
887 int bysignal = 0;
888 listNode *ln;
889 sentinelScriptJob *sj;
890
891 if (WIFSIGNALED(statloc)(((signed char) (((statloc) & 0x7f) + 1) >> 1) >
0)) bysignal = WTERMSIG(statloc)((statloc) & 0x7f);
892 sentinelEvent(LL_DEBUG0,"-script-child",NULL((void*)0),"%ld %d %d",
893 (long)pid, exitcode, bysignal);
894
895 ln = sentinelGetScriptListNodeByPid(pid);
896 if (ln == NULL((void*)0)) {
897 serverLog(LL_WARNING,"waitpid() returned a pid (%ld) we can't find in our scripts execution queue!", (long)pid)do { if (((3)&0xff) < server.verbosity) break; _serverLog
(3, "waitpid() returned a pid (%ld) we can't find in our scripts execution queue!"

898 continue;
899 }
900 sj = ln->value;
901
902 /* If the script was terminated by a signal or returns an
903 * exit code of "1" (that means: please retry), we reschedule it
904 * if the max number of retries is not already reached. */
905 if ((bysignal || exitcode == 1) &&
906 sj->retry_num != SENTINEL_SCRIPT_MAX_RETRY10)
907 {
908 sj->flags &= ~SENTINEL_SCRIPT_RUNNING1;
909 sj->pid = 0;

911 sentinelScriptRetryDelay(sj->retry_num);
912 } else {

914 * execution did not terminated in the best of the ways. */
915 if (bysignal || exitcode != 0) {
916 sentinelEvent(LL_WARNING3,"-script-error",NULL((void*)0),
917 "%s %d %d", sj->argv[0], bysignal, exitcode);
918 }
919 listDelNode(sentinel.scripts_queue,ln);
920 sentinelReleaseScriptJob(sj);
921 }
922 sentinel.running_scripts--;
923 }

1604 * and It can add old master 1 more slave.

1629 master->o_down_since_time = 0;
1630 master->s_down_since_time = 0;
1631
1632 /* Add slaves back. */
1633 for (j = 0; j < numslaves; j++) {
1634 sentinelRedisInstance *slave;
1635

1637 slaves[j]->port, master->quorum, master);
1638 releaseSentinelAddr(slaves[j]);

1640 }

1642

1644 * gets the master->addr->ip and master->addr->port as arguments. */
1645 releaseSentinelAddr(oldaddr);
1646 sentinelFlushConfig();

1648}

1809 * pre_monitor_cfg, monitor_cfg and post_monitor_cfg list */
1810void loadSentinelConfigFromQueue(void) {
1811 const char *err = NULL((void*)0);
1812 listIter li;
1813 listNode *ln;
1814 int linenum = 0;
1815 sds line = NULL((void*)0);
1816 unsigned int j;
1817
1818 /* if there is no sentinel_config entry, we can return immediately */
1819 if (server.sentinel_config == NULL((void*)0)) return;
1820
1821 list *sentinel_configs[3] = {
1822 server.sentinel_config->pre_monitor_cfg,
1823 server.sentinel_config->monitor_cfg,
1824 server.sentinel_config->post_monitor_cfg
1825 };
1826 /* loading from pre monitor config queue first to avoid dependency issues
1827 * loading from monitor config queue
1828 * loading from the post monitor config queue */
1829 for (j = 0; j < sizeof(sentinel_configs) / sizeof(sentinel_configs[0]); j++) {
1830 listRewind(sentinel_configs[j],&li);
1831 while((ln = listNext(&li))) {
1832 struct sentinelLoadQueueEntry *entry = ln->value;
1833 err = sentinelHandleConfiguration(entry->argv,entry->argc);

1835 linenum = entry->linenum;
1836 line = entry->line;
1837 goto loaderr;
1838 }
1839 }
1840 }

2682 (ri->master->failover_state ==
2683 SENTINEL_FAILOVER_STATE_WAIT_PROMOTION4))
2684 {
2685 /* Now that we are sure the slave was reconfigured as a master
2686 * set the master configuration epoch to the epoch we won the
2687 * election to perform this failover. This will force the other
2688 * Sentinels to update their config (assuming there is not
2689 * a newer one already available). */
2690 ri->master->config_epoch = ri->master->failover_epoch;
2691 ri->master->failover_state = SENTINEL_FAILOVER_STATE_RECONF_SLAVES5;
2692 ri->master->failover_state_change_time = mstime();
2693 sentinelFlushConfig();
2694 sentinelEvent(LL_WARNING3,"+promoted-slave",ri,"%@");
2695 if (sentinel.simfailure_flags &
2696 SENTINEL_SIMFAILURE_CRASH_AFTER_PROMOTION(1<<1))
2697 sentinelSimFailureCrash();
2698 sentinelEvent(LL_WARNING3,"+failover-state-reconf-slaves",
2699 ri->master,"%@");
2700 sentinelCallClientReconfScript(ri->master,SENTINEL_LEADER(1<<17),
2701 "start",ri->master->addr,ri->addr);
2702 sentinelForceHelloUpdateForMaster(ri->master);
2703 } else {
2704 /* A slave turned into a master. We want to force our view and
2705 * reconfigure as slave. Wait some time after the change before
2706 * going forward, to receive new configs if any. */
2707 mstime_t wait_time = sentinel_publish_period*4;
2708
2709 if (!(ri->flags & SRI_PROMOTED(1<<7)) &&
2710 sentinelMasterLooksSane(ri->master) &&
2711 sentinelRedisInstanceNoDownFor(ri,wait_time) &&
2712 mstime() - ri->role_reported_time > wait_time)
2713 {
2714 int retval = sentinelSendSlaveOf(ri,ri->master->addr);
2715 if (retval == C_OK0)
2716 sentinelEvent(LL_NOTICE2,"+convert-to-slave",ri,"%@");
2717 }
2718 }
2719 }
2720
2721 /* Handle slaves replicating to a different master address. */
2722 if ((ri->flags & SRI_SLAVE(1<<1)) &&
2723 role == SRI_SLAVE(1<<1) &&
2724 (ri->slave_master_port != ri->master->addr->port ||
2725 !sentinelAddrEqualsHostname(ri->master->addr, ri->slave_master_host)))
2726 {
2727 mstime_t wait_time = ri->master->failover_timeout;
2728
2729 /* Make sure the master is sane before reconfiguring this instance
2730 * into a slave. */
2731 if (sentinelMasterLooksSane(ri->master) &&
2732 sentinelRedisInstanceNoDownFor(ri,wait_time) &&
2733 mstime() - ri->slave_conf_change_time > wait_time)
2734 {
2735 int retval = sentinelSendSlaveOf(ri,ri->master->addr);
2736 if (retval == C_OK0)
2737 sentinelEvent(LL_NOTICE2,"+fix-slave-config",ri,"%@");
2738 }
2739 }
2740
2741 /* Detect if the slave that is in the process of being reconfigured
2742 * changed state. */
2743 if ((ri->flags & SRI_SLAVE(1<<1)) && role == SRI_SLAVE(1<<1) &&
2744 (ri->flags & (SRI_RECONF_SENT(1<<8)|SRI_RECONF_INPROG(1<<9))))
2745 {
2746 /* SRI_RECONF_SENT -> SRI_RECONF_INPROG. */
2747 if ((ri->flags & SRI_RECONF_SENT(1<<8)) &&
2748 ri->slave_master_host &&
2749 sentinelAddrEqualsHostname(ri->master->promoted_slave->addr,
2750 ri->slave_master_host) &&
2751 ri->slave_master_port == ri->master->promoted_slave->addr->port)
2752 {
2753 ri->flags &= ~SRI_RECONF_SENT(1<<8);
2754 ri->flags |= SRI_RECONF_INPROG(1<<9);
2755 sentinelEvent(LL_NOTICE2,"+slave-reconf-inprog",ri,"%@");
2756 }
2757
2758 /* SRI_RECONF_INPROG -> SRI_RECONF_DONE */
2759 if ((ri->flags & SRI_RECONF_INPROG(1<<9)) &&
2760 ri->slave_master_link_status == SENTINEL_MASTER_LINK_STATUS_UP0)
2761 {
2762 ri->flags &= ~SRI_RECONF_INPROG(1<<9);
2763 ri->flags |= SRI_RECONF_DONE(1<<10);
2764 sentinelEvent(LL_NOTICE2,"+slave-reconf-done",ri,"%@");
2765 }

2853/* Process a hello message received via Pub/Sub in master or slave instance,
2854 * or sent directly to this sentinel via the (fake) PUBLISH command of Sentinel.
2855 *
2856 * If the master name specified in the message is not known, the message is
2857 * discarded. */
2858void sentinelProcessHelloMessage(char *hello, int hello_len) {
2859 /* Format is composed of 8 tokens:
2860 * 0=ip,1=port,2=runid,3=current_epoch,4=master_name,
2861 * 5=master_ip,6=master_port,7=master_config_epoch. */
2862 int numtokens, port, removed, master_port;
2863 uint64_t current_epoch, master_config_epoch;
2864 char **token = sdssplitlen(hello, hello_len, ",", 1, &numtokens);
2865 sentinelRedisInstance *si, *master;
2866
2867 if (numtokens == 8) {
2868 /* Obtain a reference to the master this hello message is about */
2869 master = sentinelGetMasterByName(token[4]);
2870 if (!master) goto cleanup; /* Unknown master, skip the message. */
2871
2872 /* First, try to see if we already have this sentinel. */
2873 port = atoi(token[1]);
2874 master_port = atoi(token[6]);
2875 si = getSentinelRedisInstanceByAddrAndRunID(
2876 master->sentinels,token[0],port,token[2]);
2877 current_epoch = strtoull(token[3],NULL((void*)0),10);
2878 master_config_epoch = strtoull(token[7],NULL((void*)0),10);
2879
2880 if (!si) {
2881 /* If not, remove all the sentinels that have the same runid
2882 * because there was an address change, and add the same Sentinel
2883 * with the new address back. */
2884 removed = removeMatchingSentinelFromMaster(master,token[2]);
2885 if (removed) {
2886 sentinelEvent(LL_NOTICE2,"+sentinel-address-switch",master,
2887 "%@ ip %s port %d for %s", token[0],port,token[2]);
2888 } else {
2889 /* Check if there is another Sentinel with the same address this
2890 * new one is reporting. What we do if this happens is to set its
2891 * port to 0, to signal the address is invalid. We'll update it
2892 * later if we get an HELLO message. */
2893 sentinelRedisInstance *other =
2894 getSentinelRedisInstanceByAddrAndRunID(
2895 master->sentinels, token[0],port,NULL((void*)0));
2896 if (other) {
2897 /* If there is already other sentinel with same address (but
2898 * different runid) then remove the old one across all masters */
2899 sentinelEvent(LL_NOTICE2,"+sentinel-invalid-addr",other,"%@");
2900 dictIterator *di;
2901 dictEntry *de;
2902
2903 /* Keep a copy of runid. 'other' about to be deleted in loop. */
2904 sds runid_obsolete = sdsnew(other->runid);
2905
2906 di = dictGetIterator(sentinel.masters);
2907 while((de = dictNext(di)) != NULL((void*)0)) {
2908 sentinelRedisInstance *master = dictGetVal(de)((de)->v.val);
2909 removeMatchingSentinelFromMaster(master, runid_obsolete);
2910 }
2911 dictReleaseIterator(di);
2912 sdsfree(runid_obsolete);
2913 }
2914 }
2915
2916 /* Add the new sentinel. */
2917 si = createSentinelRedisInstance(token[2],SRI_SENTINEL(1<<2),
2918 token[0],port,master->quorum,master);
2919
2920 if (si) {
2921 if (!removed) sentinelEvent(LL_NOTICE2,"+sentinel",si,"%@");
2922 /* The runid is NULL after a new instance creation and
2923 * for Sentinels we don't have a later chance to fill it,
2924 * so do it now. */
2925 si->runid = sdsnew(token[2]);
2926 sentinelTryConnectionSharing(si);
2927 if (removed) sentinelUpdateSentinelAddressInAllMasters(si);
2928 sentinelFlushConfig();
2929 }
2930 }
2931
2932 /* Update local current_epoch if received current_epoch is greater.*/
2933 if (current_epoch > sentinel.current_epoch) {
2934 sentinel.current_epoch = current_epoch;
2935 sentinelFlushConfig();
2936 sentinelEvent(LL_WARNING3,"+new-epoch",master,"%llu",
2937 (unsigned long long) sentinel.current_epoch);
2938 }
2939
2940 /* Update master info if received configuration is newer. */
2941 if (si && master->config_epoch < master_config_epoch) {
2942 master->config_epoch = master_config_epoch;
2943 if (master_port != master->addr->port ||
2944 !sentinelAddrEqualsHostname(master->addr, token[5]))
2945 {
2946 sentinelAddr *old_addr;
2947
2948 sentinelEvent(LL_WARNING3,"+config-update-from",si,"%@");
2949 sentinelEvent(LL_WARNING3,"+switch-master",
2950 master,"%s %s %d %s %d",
2951 master->name,
2952 announceSentinelAddr(master->addr), master->addr->port,
2953 token[5], master_port);
2954
2955 old_addr = dupSentinelAddr(master->addr);
2956 sentinelResetMasterAndChangeAddress(master, token[5], master_port);
2957 sentinelCallClientReconfScript(master,
2958 SENTINEL_OBSERVER(1<<18),"start",
2959 old_addr,master->addr);
2960 releaseSentinelAddr(old_addr);
2961 }
2962 }
2963
2964 /* Update the state of the Sentinel. */
2965 if (si) si->last_hello_time = mstime();
2966 }
2967
2968cleanup:
2969 sdsfreesplitres(token,numtokens);
2970}

3003/* Send a "Hello" message via Pub/Sub to the specified 'ri' Redis
3004 * instance in order to broadcast the current configuration for this
3005 * master, and to advertise the existence of this Sentinel at the same time.
3006 *
3007 * The message has the following format:
3008 *
3009 * sentinel_ip,sentinel_port,sentinel_runid,current_epoch,
3010 * master_name,master_ip,master_port,master_config_epoch.
3011 *
3012 * Returns C_OK if the PUBLISH was queued correctly, otherwise
3013 * C_ERR is returned. */
3014int sentinelSendHello(sentinelRedisInstance *ri) {
3015 char ip[NET_IP_STR_LEN46];
3016 char payload[NET_IP_STR_LEN46+1024];
3017 int retval;
3018 char *announce_ip;
3019 int announce_port;
3020 sentinelRedisInstance *master = (ri->flags & SRI_MASTER(1<<0)) ? ri : ri->master;
3021 sentinelAddr *master_addr = sentinelGetCurrentMasterAddress(master);
3022
3023 if (ri->link->disconnected) return C_ERR-1;
3024
3025 /* Use the specified announce address if specified, otherwise try to
3026 * obtain our own IP address. */
3027 if (sentinel.announce_ip) {
3028 announce_ip = sentinel.announce_ip;
3029 } else {
3030 if (anetFdToString(ri->link->cc->c.fd,ip,sizeof(ip),NULL((void*)0),FD_TO_SOCK_NAME1) == -1)
3031 return C_ERR-1;
3032 announce_ip = ip;
3033 }
3034 if (sentinel.announce_port) announce_port = sentinel.announce_port;
3035 else if (server.tls_replication && server.tls_port) announce_port = server.tls_port;
3036 else announce_port = server.port;
3037
3038 /* Format and send the Hello message. */
3039 snprintf(payload,sizeof(payload),
3040 "%s,%d,%s,%llu," /* Info about this sentinel. */
3041 "%s,%s,%d,%llu", /* Info about current master. */
3042 announce_ip, announce_port, sentinel.myid,
3043 (unsigned long long) sentinel.current_epoch,
3044 /* --- */
3045 master->name,announceSentinelAddr(master_addr),master_addr->port,
3046 (unsigned long long) master->config_epoch);
3047 retval = redisAsyncCommand(ri->link->cc,
3048 sentinelPublishReplyCallback, ri, "%s %s %s",
3049 sentinelInstanceMapCommand(ri,"PUBLISH"),
3050 SENTINEL_HELLO_CHANNEL"__sentinel__:hello",payload);
3051 if (retval != C_OK0) return C_ERR-1;
3052 ri->link->pending_commands++;
3053 return C_OK0;
3054}

3111/* Send periodic PING, INFO, and PUBLISH to the Hello channel to
3112 * the specified master or slave instance. */
3113void sentinelSendPeriodicCommands(sentinelRedisInstance *ri) {
3114 mstime_t now = mstime();
3115 mstime_t info_period, ping_period;
3116 int retval;
3117
3118 /* Return ASAP if we have already a PING or INFO already pending, or
3119 * in the case the instance is not properly connected. */
3120 if (ri->link->disconnected) return;
3121
3122 /* For INFO, PING, PUBLISH that are not critical commands to send we
3123 * also have a limit of SENTINEL_MAX_PENDING_COMMANDS. We don't
3124 * want to use a lot of memory just because a link is not working
3125 * properly (note that anyway there is a redundant protection about this,
3126 * that is, the link will be disconnected and reconnected if a long
3127 * timeout condition is detected. */
3128 if (ri->link->pending_commands >=
3129 SENTINEL_MAX_PENDING_COMMANDS100 * ri->link->refcount) return;
3130
3131 /* If this is a slave of a master in O_DOWN condition we start sending
3132 * it INFO every second, instead of the usual SENTINEL_INFO_PERIOD
3133 * period. In this state we want to closely monitor slaves in case they
3134 * are turned into masters by another Sentinel, or by the sysadmin.
3135 *
3136 * Similarly we monitor the INFO output more often if the slave reports
3137 * to be disconnected from the master, so that we can have a fresh
3138 * disconnection time figure. */
3139 if ((ri->flags & SRI_SLAVE(1<<1)) &&
3140 ((ri->master->flags & (SRI_O_DOWN(1<<4)|SRI_FAILOVER_IN_PROGRESS(1<<6))) ||
3141 (ri->master_link_down_time != 0)))
3142 {
3143 info_period = 1000;
3144 } else {
3145 info_period = sentinel_info_period;
3146 }
3147
3148 /* We ping instances every time the last received pong is older than
3149 * the configured 'down-after-milliseconds' time, but every second
3150 * anyway if 'down-after-milliseconds' is greater than 1 second. */
3151 ping_period = ri->down_after_period;
3152 if (ping_period > sentinel_ping_period) ping_period = sentinel_ping_period;
3153
3154 /* Send INFO to masters and slaves, not sentinels. */
3155 if ((ri->flags & SRI_SENTINEL(1<<2)) == 0 &&
3156 (ri->info_refresh == 0 ||
3157 (now - ri->info_refresh) > info_period))
3158 {
3159 retval = redisAsyncCommand(ri->link->cc,
3160 sentinelInfoReplyCallback, ri, "%s",
3161 sentinelInstanceMapCommand(ri,"INFO"));
3162 if (retval == C_OK0) ri->link->pending_commands++;
3163 }
3164
3165 /* Send PING to all the three kinds of instances. */
3166 if ((now - ri->link->last_pong_time) > ping_period &&
3167 (now - ri->link->last_ping_time) > ping_period/2) {
3168 sentinelSendPing(ri);
3169 }
3170
3171 /* PUBLISH hello messages to all the three kinds of instances. */
3172 if ((now - ri->last_pub_time) > sentinel_publish_period) {
3173 sentinelSendHello(ri);
3174 }
3175}

4887/* Select a suitable slave to promote. The current algorithm only uses
4888 * the following parameters:
4889 *
4890 * 1) None of the following conditions: S_DOWN, O_DOWN, DISCONNECTED.
4891 * 2) Last time the slave replied to ping no more than 5 times the PING period.
4892 * 3) info_refresh not older than 3 times the INFO refresh period.
4893 * 4) master_link_down_time no more than:
4894 * (now - master->s_down_since_time) + (master->down_after_period * 10).
4895 * Basically since the master is down from our POV, the slave reports
4896 * to be disconnected no more than 10 times the configured down-after-period.
4897 * This is pretty much black magic but the idea is, the master was not
4898 * available so the slave may be lagging, but not over a certain time.
4899 * Anyway we'll select the best slave according to replication offset.
4900 * 5) Slave priority can't be zero, otherwise the slave is discarded.
4901 *
4902 * Among all the slaves matching the above conditions we select the slave
4903 * with, in order of sorting key:
4904 *
4905 * - lower slave_priority.
4906 * - bigger processed replication offset.
4907 * - lexicographically smaller runid.
4908 *
4909 * Basically if runid is the same, the slave that processed more commands
4910 * from the master is selected.
4911 *
4912 * The function returns the pointer to the selected slave, otherwise
4913 * NULL if no suitable slave was found.
4914 */
4915
4916/* Helper for sentinelSelectSlave(). This is used by qsort() in order to
4917 * sort suitable slaves in a "better first" order, to take the first of
4918 * the list. */
4919int compareSlavesForPromotion(const void *a, const void *b) {
4920 sentinelRedisInstance **sa = (sentinelRedisInstance **)a,
4921 **sb = (sentinelRedisInstance **)b;
4922 char *sa_runid, *sb_runid;
4923
4924 if ((*sa)->slave_priority != (*sb)->slave_priority)
4925 return (*sa)->slave_priority - (*sb)->slave_priority;
4926
4927 /* If priority is the same, select the slave with greater replication
4928 * offset (processed more data from the master). */
4929 if ((*sa)->slave_repl_offset > (*sb)->slave_repl_offset) {
4930 return -1; /* a < b */
4931 } else if ((*sa)->slave_repl_offset < (*sb)->slave_repl_offset) {
4932 return 1; /* a > b */
4933 }
4934
4935 /* If the replication offset is the same select the slave with that has
4936 * the lexicographically smaller runid. Note that we try to handle runid
4937 * == NULL as there are old Redis versions that don't publish runid in
4938 * INFO. A NULL runid is considered bigger than any other runid. */
4939 sa_runid = (*sa)->runid;
4940 sb_runid = (*sb)->runid;
4941 if (sa_runid == NULL((void*)0) && sb_runid == NULL((void*)0)) return 0;
4942 else if (sa_runid == NULL((void*)0)) return 1; /* a > b */
4943 else if (sb_runid == NULL((void*)0)) return -1; /* a < b */
4944 return strcasecmp(sa_runid, sb_runid);
4945}

5324/* This function checks if we need to enter the TILT mode.
5325 *
5326 * The TILT mode is entered if we detect that between two invocations of the
5327 * timer interrupt, a negative amount of time, or too much time has passed.
5328 * Note that we expect that more or less just 100 milliseconds will pass
5329 * if everything is fine. However we'll see a negative number or a
5330 * difference bigger than SENTINEL_TILT_TRIGGER milliseconds if one of the
5331 * following conditions happen:
5332 *
5333 * 1) The Sentinel process for some time is blocked, for every kind of
5334 * random reason: the load is huge, the computer was frozen for some time
5335 * in I/O or alike, the process was stopped by a signal. Everything.
5336 * 2) The system clock was altered significantly.
5337 *
5338 * Under both this conditions we'll see everything as timed out and failing
5339 * without good reasons. Instead we enter the TILT mode and wait
5340 * for SENTINEL_TILT_PERIOD to elapse before starting to act again.
5341 *
5342 * During TILT time we still collect information, we just do not act. */
5343void sentinelCheckTiltCondition(void) {
5344 mstime_t now = mstime();
5345 mstime_t delta = now - sentinel.previous_time;
5346
5347 if (delta < 0 || delta > sentinel_tilt_trigger) {
